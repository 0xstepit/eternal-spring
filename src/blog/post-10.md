# Compiling solidity contracts

Given a smart contract implementation in Solidity named `WEVMOS9TestCaller.sol`:

```solidity
// SPDX-License-Identifier: LGPL-3.0-only
pragma solidity >=0.8.17;

import "../IWERC20.sol" as werc20Precompile;

contract WEVMOS9TestCaller {
    werc20Precompile.IWERC20 public wrappedToken;
    uint256 public counter;

    constructor(address payable _wrappedTokenAddress) {
        wrappedToken = werc20Precompile.IWERC20(_wrappedTokenAddress);
        counter = 0;
    }

    function depositWithRevert(bool before, bool aft) public payable {
        counter++;

        uint amountIn = msg.value;
        wrappedToken.deposit{value: amountIn}();

        if (before) {
            require(false, "revert here");
        }

        counter--;

        if (aft) {
            require(false, "revert here");
        }
        return;
    }
}
```

We can compile it with **solc** to get the opcodes used by the contract:

```sh
solc WEVMOS9TestCaller.sol --base-path . --include-path .. --include-path ../../ --opcodes > opppagamgam.txt
```
